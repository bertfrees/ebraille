eBraille styling

The examples in this section assume that eBraille reading systems do not use the same default style sheets that web browsers do. To try these examples in a browser, make sure the following style sheet is applied and the same monospaced font and font size is used for all text:

* {
   font-weight: normal;
   line-height: 1rem;
   letter-spacing: 0;
   margin: 0;
   padding: 0;
   border: none;
   list-style: none;
   color: black;
   text-decoration: none;
}

the following CSS style sheet may be applied:

We make use of the non-standard CSS pseudo-class :-moz-first-node in order to be able to style a .linenum without preceding text differently (because there must not be a gap of three blank cells when the first print line begins a new braille line). 

@property --widest-linenum {
   syntax: <length>;
}
.prose {
   /* By default, i.e. without scripting, line numbers are assumed to only be two
      characters wide, and a column of four characters is preserved for the
      numbers. The actual value of --widest-linenum can be determined and set from
      JavaScript. Note that, regardless of the width of the column, each line
      number is separated from the text by at least two blanks. */
   --widest-linenum: 2ch;
   padding-right: calc(var(--widest-linenum) + 2ch);
}
.prose .linenum:not(:-moz-first-node)::before {
   /* The pseudo element should not take up space when at the end of the line. If
      we use a no-break space (" \A0 "), it does take up space and as a result
      might be moved to the next line. If we use a normal space with
      `white-space: pre-wrap', it does not take up space, however then we can't
      specify that the spaces at the beginning and end should be collapsible. This
      can be fixed with scripting. When scripting is disabled, we choose a
      possible gap at the beginning of the line over a possibly too wide gap. Note
      that we can get a dispositioned line number in either case. */
   content: " \A0 ";
}
.prose .linenum::after {
   content: attr(aria-label);
   float: right;
   clear: right;
   margin-left: 2ch;
   margin-right: -4ch;
}

/**
 * CSS rules belonging to the JavaScript code below
 */
.prose .linenum-anchor::before {

}

.prose .linenum-anchor + br + .linenum:not(:-moz-first-node)::before {
   content: "​\A0";

and the following JavaScript may be applied to enhance the styling:

This example shows possible use cases for allowing scripting in the future. Because scripting is currenly forbidden, we obviously shouldn't include it in the best practices. Still it seems useful to have it included somewhere, to show the limitations of CSS. 

window.onload = function() {
  document.querySelectorAll(".prose").forEach(prose => {
    let linenums = Array.from(prose.querySelectorAll(".linenum"));
    // find widest line number and set width of right margin
    let maxWidth = linenums.reduce(
      (max, linenum) => Math.max(max, linenum.getAttribute("aria-label").length),
      0);
    prose.style.setProperty("--widest-linenum", maxWidth + "ch");
    let prevPosition = -1000;
    linenums.forEach(linenum => {
      // insert an extra span before the .linenum to have more control over line
      // breaking and positioning of number and gap
      let precedingSibling = linenum.previousSibling;
      while (precedingSibling) {
        if (precedingSibling.nodeType === 8); // comment
        else if (precedingSibling.nodeType !== 3  // not text
                 || precedingSibling.nodeValue.trim() !== '') {
          let anchor = linenum.parentNode.insertBefore(document.createElement("span"), linenum);
          anchor.className = "linenum-anchor";
          break;
        }
        precedingSibling = precedingSibling.previousSibling;
      }
      // trim white space after to make .linenum::before stick to following word,
      // and to make sure gaps are always exactly 3 cells wide
      let followingSibling = linenum.nextSibling;
      while (followingSibling) {
        if (followingSibling.nodeType === 8); // comment
        else if (followingSibling.nodeType === 3) { // text
          let trimmed = followingSibling.nodeValue.replace(/^\s+/, "");
          followingSibling.nodeValue = trimmed;
          if (trimmed !== '')
            break;
        } else
          break;
        followingSibling = followingSibling.nextSibling;
      }
      // if the following text is on the same braille line as the braille line on
      // which the previous print line started, insert a <br/>
      if (followingSibling.nodeType === 3) {
        let range = document.createRange();
        range.selectNode(followingSibling);
        let pos = range.getBoundingClientRect().top;
        range.detach();
        if (pos <= prevPosition)
          linenum.parentNode.insertBefore(document.createElement("br"), linenum);
        prevPosition = pos;
      }
    });
  });
};

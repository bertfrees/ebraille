In order to ensure columns are aligned to the grid, the column widths must be set explicitly. Experience shows that setting column widths can be tricky, and even if it works flawlessly, it is not very practical. 

table {
   display: table;
   table-layout: fixed;
   border-collapse: separate;
   border-spacing: 0;
   border-width: .4em 0;
   border-style: solid;
   border-color: black;
   margin: .2em 0 .3em 0;
}
thead tr th {
   text-align: left;
   padding-top: .3em;
   padding-bottom: .4em;
   border-bottom: .1em solid black;
}
tbody tr td {
   padding-top: .4em;
}
tbody tr + tr td {
   padding-top: 0;
   padding-bottom: .2em;
}
tbody tr + tr:has(+ tr) td {
   padding-bottom: 0;
}
col {
   width: 70px;
}
col + col {
   width: 140px;
}
col + col + col {
   width: 90px;
}
col + col + col + col {
   width: 100px;
}
th, td {
   padding-left: 0;
   padding-right: 0;
   vertical-align: top;
}
th:has(+ th), td:has(+ td) {
   padding-right: 1ch;
}
